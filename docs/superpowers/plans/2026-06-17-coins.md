```gdscript
func _collect_coins() -> void:
	for coin in coins:
		if bool(coin.get("collected", false)):
			continue
		if absf(float(coin["z"])) > COIN_COLLECTION_Z_RADIUS:
			continue
		if absf(current_x - lane_x(int(coin["lane"]))) > COIN_COLLECTION_X_RADIUS:
```

```gdscript
func _uses_coin_hud_and_persistence() -> bool:
	var coin_save_path := "user://dog_run_coin_smoke.cfg"
	var absolute_path := ProjectSettings.globalize_path(coin_save_path)
	DirAccess.remove_absolute(absolute_path)
	_main._save_path = coin_save_path

	var coin_label: Label = _main.find_child("CoinLabel", true, false)
	if coin_label == null:
		DirAccess.remove_absolute(absolute_path)
		return false
	var hud_font: FontFile = _main.get("_hud_font")
	if coin_label.get_theme_font("font") != hud_font:
		DirAccess.remove_absolute(absolute_path)
		return false
	if coin_label.horizontal_alignment != HORIZONTAL_ALIGNMENT_LEFT:
		DirAccess.remove_absolute(absolute_path)
		return false

	_main._total_coins = 12
	_main.simulation.state = _main.simulation.RunState.READY
	_main._update_hud()
	if not coin_label.visible or coin_label.text != "COINS 12":
		DirAccess.remove_absolute(absolute_path)
		return false
	_main.simulation.state = _main.simulation.RunState.RUNNING
	_main._update_hud()
	if not coin_label.visible:
		DirAccess.remove_absolute(absolute_path)
		return false
	_main.simulation.state = _main.simulation.RunState.GAME_OVER
	_main._update_hud()
	if not coin_label.visible:
		DirAccess.remove_absolute(absolute_path)
		return false

	_main._save_progress()
	_main._total_coins = 0
	_main._load_progress()
	if _main._total_coins != 12:
		DirAccess.remove_absolute(absolute_path)
		return false

	_main._handle_coin_events([{"type": "coin_collected", "coin_id": 777}])
	var config := ConfigFile.new()
	var persisted := config.load(coin_save_path) == OK
	persisted = persisted and int(config.get_value("currency", "coins", 0)) == 13
	persisted = persisted and _main._total_coins == 13
	_main.simulation.state = _main.simulation.RunState.RUNNING
	_main._update_hud()
	DirAccess.remove_absolute(absolute_path)
	return persisted
```

```gdscript
func _uses_coin_visuals() -> bool:
	if _main.simulation.coins.is_empty():
		return false
	_main._sync_coins(0.1)
	if _main._coin_nodes.is_empty():
		return false
	var coin_node: MeshInstance3D = _main._coin_nodes.values()[0]
	if coin_node == null or not coin_node.visible:
		return false
	if not coin_node.mesh is CylinderMesh:
		return false
	if not coin_node.material_override is StandardMaterial3D:
		return false
	var material: StandardMaterial3D = coin_node.material_override
	if not material.emission_enabled:
		return false
	if material.albedo_color.r < 0.9 or material.albedo_color.g < 0.6:
		return false
	var before_rotation := coin_node.rotation.y
	_main._sync_coins(0.2)
	return not is_equal_approx(coin_node.rotation.y, before_rotation)
```

```gdscript
func _sync_coins(delta: float) -> void:
	var active_ids := {}
	for coin in simulation.coins:
		if bool(coin.get("collected", false)):
			continue
		var coin_id: int = coin["id"]
		active_ids[coin_id] = true

		var mesh_instance: MeshInstance3D
		if _coin_nodes.has(coin_id):
			mesh_instance = _coin_nodes[coin_id]
		else:
			mesh_instance = _acquire_coin()
			if mesh_instance == null:
				continue
			_coin_nodes[coin_id] = mesh_instance

		mesh_instance.position = Vector3(
			simulation.lane_x(int(coin["lane"])),
			0.75 + float(coin["height"]),
			float(coin["z"]) + TUNING.visual_action_plane_z
		)
		mesh_instance.rotation.x = PI * 0.5
		mesh_instance.rotation.y = fposmod(mesh_instance.rotation.y + COIN_ROTATION_SPEED * delta, TAU)
		mesh_instance.visible = true

	for coin_id in _coin_nodes.keys():
		if active_ids.has(coin_id):
			continue
		var mesh_instance: MeshInstance3D = _coin_nodes[coin_id]
		_coin_nodes.erase(coin_id)
		mesh_instance.visible = false
		_coin_pool.append(mesh_instance)


func _acquire_coin() -> MeshInstance3D:
	if not _coin_pool.is_empty():
		return _coin_pool.pop_back()
	if _coin_nodes.size() + _coin_pool.size() >= MAX_COIN_NODES:
		return null
	var mesh_instance := MeshInstance3D.new()
	mesh_instance.name = "Coin"
	_configure_coin(mesh_instance)
	add_child(mesh_instance)
	return mesh_instance


func _configure_coin(mesh_instance: MeshInstance3D) -> void:
	var coin_mesh := CylinderMesh.new()
	coin_mesh.top_radius = COIN_RADIUS
	coin_mesh.bottom_radius = COIN_RADIUS
	coin_mesh.height = COIN_THICKNESS
	coin_mesh.radial_segments = 32
	coin_mesh.rings = 1
	mesh_instance.mesh = coin_mesh
	mesh_instance.material_override = _make_coin_material()
```
